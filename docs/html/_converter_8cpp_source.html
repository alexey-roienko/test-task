88 bool Converter::findRegion() {
89  using namespace cv;
90  using namespace std;
91 
92  // The first pixel is always TRUE and is painted by white color
93  maskImg.at<uchar>(pxPos.y, pxPos.x) = 255;
94  // Queue for storing the pixels to be checked
95  queue<Point> fifo;
96 
97  // Lambda-function for checking the current pixel
98  auto lPxCheck = [&fifo, this](int pxX, int pxY)->bool{
99  //if (this->debug)
100  //cout << "lPxCheck: pxRow = " << pxY << ", pxCol = " << pxX << endl;
101  // Perform checking
102  if (checkPxPos(pxY, pxX) && checkPxMask(pxY, pxX) && checkPxColor(pxY, pxX)) {
103  fifo.push(Point(pxX, pxY));
104  return true;
105  }
106  return false;
107  };
108 
109  // Push the initial pixels located on cross with the center in 'pixel'
110  int X = pxPos.x;
111  int Y = pxPos.y-1;
112  lPxCheck(X, Y);
113  X = pxPos.x+1; Y = pxPos.y; lPxCheck(X, Y);
114  X = pxPos.x; Y = pxPos.y+1; lPxCheck(X, Y);
115  X = pxPos.x-1; Y = pxPos.y; lPxCheck(X, Y);
116 
117  while (not fifo.empty()) {
118  Point px2Check = fifo.front();
119  // Checking the first neighbour pixel
120  X = px2Check.x; Y = px2Check.y-1; lPxCheck(X, Y);
121  // Checking the second neighbour pixel
122  X = px2Check.x+1; Y = px2Check.y; lPxCheck(X, Y);
123  // Checking the third neighbour pixel
124  X = px2Check.x; Y = px2Check.y+1; lPxCheck(X, Y);
125  // Checking the fourth neighbour pixel
126  X = px2Check.x-1; Y = px2Check.y; lPxCheck(X, Y);
127  // Remove the checked point
128  fifo.pop();
129  }
130  return 0;
131 }

188 double Converter::calcPixelDistance(const cv::Vec3b & px2Comp, const cv::Vec3b & initPx, uint8_t distType) {
189  using namespace std;
190 
191  double dist = 0;
192  switch (distType) {
193  // Simple BRG distance
194  case 0: {
195  dist = (px2Comp[2]-initPx[2])*(px2Comp[2]-initPx[2]) +
196  (px2Comp[1]-initPx[1])*(px2Comp[1]-initPx[1]) +
197  (px2Comp[0]-initPx[0])*(px2Comp[0]-initPx[0]);
198  break;
199  }
200  // Simple BRG distance with coefficients
201  case 1: {
202  dist = 2*(px2Comp[2]-initPx[2])*(px2Comp[2]-initPx[2]) +
203  4*(px2Comp[1]-initPx[1])*(px2Comp[1]-initPx[1]) +
204  3*(px2Comp[0]-initPx[0])*(px2Comp[0]-initPx[0]);
205  break;
206  }
207  // Advanced BRG distance
208  case 2: {
209  double dR2 = (px2Comp[2]-initPx[2]) * (px2Comp[2]-initPx[2]);
210  double midR = (px2Comp[2] + initPx[2])*0.5;
211  double dG2 = (px2Comp[1]-initPx[1]) * (px2Comp[1]-initPx[1]);
212  double dB2 = (px2Comp[0]-initPx[0]) * (px2Comp[0]-initPx[0]);
213  dist = 2*dR2 + 4*dG2 + 3*dB2 + midR * (dR2 - dB2) * DefParams::FR256;
214  break;
215  }
216  // Simple Lab distance - CIE 76
217  case 3: {
218  double dL2 = (px2Comp[0]-initPx[0]) * (px2Comp[0]-initPx[0]);
219  double dA2 = (px2Comp[1]-initPx[1]) * (px2Comp[1]-initPx[1]);
220  double dB2 = (px2Comp[2]-initPx[2]) * (px2Comp[2]-initPx[2]);
221  dist = dL2 + dA2 + dB2;
222  break;
223  }
224  // Advanced Lab distance - CIE 94
225  case 4: {
226  double dL2 = (px2Comp[0] - initPx[0]) * (px2Comp[0] - initPx[0]);
227  double C1_2 = px2Comp[1] * px2Comp[1] + px2Comp[2] * px2Comp[2];
228  double C2_2 = initPx[1] * initPx[1] + initPx[2] * initPx[2];
229  double C1 = exp(0.5 * log(C1_2));
230  double C2 = exp(0.5 * log(C2_2));
231  double dC2 = (C1 - C2) * (C1 - C2);
232  double dA2 = (px2Comp[1] - initPx[1]) * (px2Comp[1] - initPx[1]);
233  double dB2 = (px2Comp[2] - initPx[2]) * (px2Comp[2] - initPx[2]);
234  double dH2 = dA2 + dB2 - dC2;
235  double SC = 1 + DefParams::K1 * C1;
236  double SH = 1 + DefParams::K2 * C1;
237  if ((DefParams::k_L - 1 < 1e-5) && (DefParams::SL - 1 < 1e-5))
238  dist = dL2 + dC2/(SC*SC) + dH2/(SH*SH);
239  else if (DefParams::k_L - 1 < 1e-5)
240  dist = dL2/(DefParams::SL*DefParams::SL) + dC2/(SC*SC) + dH2/(SH*SH);
241  else if (DefParams::SL - 1 < 1e-5)
242  dist = dL2/(DefParams::k_L*DefParams::k_L) + dC2/(SC*SC) + dH2/(SH*SH);
243  else
244  dist = dL2/(DefParams::k_L*DefParams::k_L)/(DefParams::SL*DefParams::SL) + dC2/(SC*SC) + dH2/(SH*SH);
245  break;
246  }
247  }
248  if (debug) cout << "\t Color distance = " << setprecision(5) << dist << endl;
249  return dist;
250 }
